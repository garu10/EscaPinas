    <script>
        function updateUI(locationName, imageName, attractions) {
            // 1. Update the Banner Image (Ensure folder path is correct)
            const banner = document.getElementById('bannerImage');
            banner.src = 'frontend/assets/images/' + imageName;

            // 2. Update the button text
            document.getElementById('locationBtn').innerText = locationName;

            // 3. Update the Attraction Dropdown
            const select = document.getElementById('attractionSelect');
            select.disabled = false;
            select.innerHTML = '<option value="">Select Attraction...</option>'; 
            
            attractions.forEach(function(item) {
                let option = document.createElement('option');
                option.value = item;
                option.text = item;
                select.appendChild(option);
            });
        }
    </script>
